cmake_minimum_required(VERSION 3.14)
project(unit_test)

set(CMAKE_CXX_STANDARD 17)

# Incluir m칩dulo FetchContent
include(FetchContent)

# Declarar GoogleTest
FetchContent_Declare(
  googletest
  URL https://github.com/google/googletest/archive/release-1.12.1.zip
)

# Prevenir la anulaci칩n de configuraciones del compilador/enlazador en Windows
set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)

# Hacer disponible GoogleTest
FetchContent_MakeAvailable(googletest)

# Fuentes del proyecto
set(SOURCES
    main.cpp
    MyMathLib.h
    MetodosFactorial.h
    MetodosOperaciones.h

)

# A침adir ejecutable para pruebas de operaciones
add_executable(test_Operaciones tests/test_Operaciones.cpp ${SOURCES})
target_link_libraries(test_Operaciones gtest gtest_main)

# A침adir ejecutable para pruebas de combinatoria
add_executable(test_Factorial tests/test_Factorial.cpp ${SOURCES})
target_link_libraries(test_Factorial gtest gtest_main)

# Registrar pruebas
include(GoogleTest)
gtest_discover_tests(test_Operaciones)
gtest_discover_tests(test_Factorial)
